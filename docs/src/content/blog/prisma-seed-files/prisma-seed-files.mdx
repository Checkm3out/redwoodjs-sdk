```ts title="multipleApplicationsSeed.ts"
import { defineScript } from "redwoodsdk/worker";
import { db, setupDb } from "../db";
import { faker } from '@faker-js/faker';

const NUM_APPLICATIONS = 10; // Adjust as needed

const createFakeApplication = () => {
  const salaryMin = faker.number.int({ min: 80000, max: 150000 }).toString();
  const salaryMax = faker.number.int({
    min: parseInt(salaryMin),
    max: parseInt(salaryMin) + 50000
  }).toString();

  return {
    user: {
      connect: {
        id: "0f9a097c-d7bc-4ab5-8b11-6942163df348",
      },
    },
    status: {
      connect: {
        id: faker.number.int({ min: 1, max: 5 }),
      },
    },
    salaryMin,
    salaryMax,
    jobTitle: faker.person.jobTitle(),
    jobDescription: faker.lorem.paragraphs(2),
    postingUrl: faker.internet.url(),
    dateApplied: faker.date.past({ years: 0.5 }),
    company: {
      create: {
        name: faker.company.name(),
        contacts: {
          create: {
            firstName: faker.person.firstName(),
            lastName: faker.person.lastName(),
            email: faker.internet.email(),
            role: faker.person.jobTitle(),
          },
        },
      },
    },
  };
};

export default defineScript(async ({ env }) => {
  setupDb(env);

  const applications = Array.from({ length: NUM_APPLICATIONS }, createFakeApplication);

  for (const application of applications) {
    await db.application.create({ data: application });
  }

  console.log(`ðŸŒ± Finished seeding ${NUM_APPLICATIONS} applications`);
});
```
